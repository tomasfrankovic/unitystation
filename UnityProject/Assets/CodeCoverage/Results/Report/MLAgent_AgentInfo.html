 212    /// environment. Observations are determined by the cameras attached

 217    /// </summary>
 218    /// <remarks>
 219    /// Simply speaking, an agent roams through an environment and at each step
 220    /// of the environment extracts its current observation, sends them to its
 221    /// linked brain and in return receives an action from its brain. In practice,
 222    /// however, an agent need not send its observation at every step since very
 223    /// little may have changed between sucessive steps. Currently, how often an
 224    /// agent updates its brain with a fresh observation is determined by the
 225    /// Academy.
 226    ///
 227    /// At any step, an agent may be considered <see cref="m_Done"/>.
 228    /// This could occur due to a variety of reasons:
 229    ///     - The agent reached an end state within its environment.
 230    ///     - The agent reached the maximum # of steps (i.e. timed out).
 231    ///     - The academy reached the maximum # of steps (forced agent to be done).
 232    ///
 233    /// Here, an agent reaches an end state if it completes its task successfully
 234    /// or somehow fails along the way. In the case where an agent is done before
 235    /// the academy, it either resets and restarts, or just lingers until the
 236    /// academy is done.

 239    /// corresponds to an academy step, which also corresponds to Unity
 240    /// environment step (i.e. each FixedUpdate call). This is not the case for
 241    /// episodes. The academy controls the global episode count and each agent

 246    /// multiple local episodes. Consequently, if an agent max step is

 578        /// <summary>
 579        /// Sends the Agent info to the linked Brain.
 580        /// </summary>
 581        void SendInfoToBrain()
 582        {
 583            if (brain == null)
 584            {
 585                return;
 586            }
 587
 588            m_Info.memories = m_Action.memories;
 589            m_Info.storedVectorActions = m_Action.vectorActions;
 590            m_Info.storedTextActions = m_Action.textActions;
 591            m_Info.vectorObservation.Clear();
 592            m_ActionMasker.ResetMask();
 593            CollectObservations();
 594            m_Info.actionMasks = m_ActionMasker.GetMask();
 595
 596            var param = brain.brainParameters;
 597            if (m_Info.vectorObservation.Count != param.vectorObservationSize)
 598            {
 599                throw new UnityAgentsException(string.Format(
 600                    "Vector Observation size mismatch between continuous " +
 601                    "agent {0} and brain {1}. " +
 602                    "Was Expecting {2} but received {3}. ",
 603                    gameObject.name, brain.name,
 604                    brain.brainParameters.vectorObservationSize,
 605                    m_Info.vectorObservation.Count));
 606            }
 607
 608            Utilities.ShiftLeft(m_Info.stackedVectorObservation, param.vectorObservationSize);
 609            Utilities.ReplaceRange(m_Info.stackedVectorObservation, m_Info.vectorObservation,
 610                m_Info.stackedVectorObservation.Count - m_Info.vectorObservation.Count);
 611
 612            m_Info.visualObservations.Clear();
 613            var visualObservationCount = agentParameters.agentCameras.Count + agentParameters.agentRenderTextures.Count;
 614            if (param.cameraResolutions.Length > visualObservationCount)
 615            {
 616                throw new UnityAgentsException(string.Format(
 617                    "Not enough cameras/renderTextures for agent {0} : Brain {1} expecting at " +
 618                    "least {2} cameras/renderTextures but only {3} were present.",
 619                    gameObject.name, brain.name,
 620                    brain.brainParameters.cameraResolutions.Length,
 621                    visualObservationCount));
 622            }
 623
 624            //First add all cameras
 625            for (var i = 0; i < agentParameters.agentCameras.Count; i++)
 626            {
 627                var obsTexture = ObservationToTexture(
 628                    agentParameters.agentCameras[i],
 629                    param.cameraResolutions[i].width,
 630                    param.cameraResolutions[i].height);
 631                m_Info.visualObservations.Add(obsTexture);
 632            }
 633
 634            //Then add all renderTextures
 635            var camCount = agentParameters.agentCameras.Count;
 636            for (var i = 0; i < agentParameters.agentRenderTextures.Count; i++)
 637            {
 638                var obsTexture = ObservationToTexture(
 639                    agentParameters.agentRenderTextures[i],
 640                    param.cameraResolutions[camCount + i].width,
 641                    param.cameraResolutions[camCount + i].height);
 642                m_Info.visualObservations.Add(obsTexture);
 643            }
 644
 645            m_Info.reward = m_Reward;
 646            m_Info.done = m_Done;
 647            m_Info.maxStepReached = m_MaxStepReached;
 648            m_Info.id = m_Id;
 649
 650            brain.SendState(this, m_Info);
 651
 652            if (m_Recorder != null && m_Recorder.record && Application.isEditor)
 653            {
 654                m_Recorder.WriteExperience(m_Info);
 655            }
 656
 657            m_Info.textObservation = "";
 658        }
